# pathToTree
path string to object tree


## use

```js
import pathToTree from './pathToTress'

var arr = [
    "0000000001/",
    "0000000001/b.png",
    "0000000001/new_b.png",
    "0000000001/watermark.zip",
    "0000000001/ass",
    "0000000001/ass/ass.txt",
    "0000000001/tes",
    "0000000001/tes/test.txt",
]

console.log(pathToTree(arr))

```
